project/
│
├── data/
│   ├── raw/                          # 원본 CWRU .mat 파일 (전부 12kHz로 통일해서 사용)
│   │   ├── normal/                   # 정상 신호 (학습용 — 원본 48k → 12k 다운샘플)
│   │   ├── fault_12k/                # 고장 12kHz (주 평가 데이터, DE)
│   │   ├── fan_end_fault/            # Fan End 고장 12kHz (평가 전용 — FE 가속도계)
│   │   └── fault_48k/                # 고장 48kHz (로더에서 12k로 다운샘플)
│   ├── processed/                    # 윈도우 슬라이싱된 신호 배열
│   ├── features/                     # 모델 입력용 스펙트로그램 .npy 배열 (수치 손실 방지)
│   ├── spectrograms/                 # 사람이 눈으로 볼 시각화 이미지 전용 (학습엔 사용 안 함)
│   └── splits/                       # train/val/test 분리 정보
│
├── configs/
│   └── config.yaml                   # 샘플링, window size, n_fft/hop, threshold, 경로 설정
│
├── src/
│   ├── __init__.py                   # src 를 패키지로 인식시키는 빈 파일
│   ├── schemas.py                    # 모든 값 데이터 타입·최종 출력 JSON 정의 (팀 공통 계약)
│   ├── data_loader.py                # .mat 파일 로드, CWRU 변수 추출 (정상/고장 분리)
│   ├── preprocessing.py              # window slicing, normalization (정상 통계로만 fit)
│   ├── spectrogram.py                # STFT/CWT 변환 (로그 dB 스케일)
│   ├── streaming.py                  # 실시간 스트리밍 시뮬레이터 + 슬라이딩 윈도우 버퍼
│   │
│   ├── baseline_statistical.py       # 경로 A: RMS, peak, kurtosis 기반 임펄스 탐지
│   ├── autoencoder_detector.py       # 경로 B: Conv Autoencoder 학습/추론 (정상만 학습, 12k)
│   ├── cnn_fault_classifier.py       # 경로 C: 위치(IR/OR/B) 분류 — 물리+CNN 교차검증 (12k)
│   │
│   ├── fusion.py                     # A+B anomaly score 융합 + 디바운스
│   ├── inference.py                  # 오케스트레이터: 전처리→A·B 병렬→융합→(이상이면)C→결과 조립
│   ├── evaluator.py                  # AUC, pAUC, FPR, confusion matrix 평가
│   └── utils.py                      # 공통 함수 (config 로드, device 선택, 시드 고정 등)
│
├── models/
│   ├── autoencoder.pth               # 학습된 AE 모델 (경로 B)
│   ├── fault_classifier.pth          # 학습된 CNN 모델 (경로 C)
│   ├── scaler.pkl                    # 정규화 통계 저장 (배포 시 동일 전처리 보장)
│   └── thresholds.json               # A·B 임계값 + 산정 근거 메타데이터 (디바운스 N 은 config)
│
├── outputs/
│   ├── figures/                      # 결과 그래프 (PNG)
│   ├── heatmaps/                     # reconstruction error map (B의 이상 근거 시각화)
│   └── reports/                      # 평가 결과 csv/txt
│
├── visualization/                    # 그래프 스크립트 (각각 독립 실행, 결과는 outputs/figures)
│   ├── _common.py                    # 폰트·예시신호 선택 (공유)
│   ├── windowing_normalization.py    # 윈도잉 + 정규화 파형
│   ├── signal_overview.py            # 정상 vs 고장 전체길이 비교
│   ├── spectrogram_comparison.py     # 스펙트로그램 비교
│   └── feature_comparison.py         # 경로 A 특징 분포
│
├── backend/                          # app/ → backend / frontend 분리 (프론트 팀 필요)
│   └── api.py                        # FastAPI: 추론 결과를 JSON 으로 제공
│
├── frontend/                         # 웹 대시보드 (프론트엔드 팀 담당, 실시간 파형·점수·알람)
│
├── tests/                            # 단위 테스트 (예: 임펄스 주입 검증 T1.3)
│
├── train_autoencoder.py              # 경로 B 학습 실행 파일
├── train_classifier.py               # 경로 C 학습 실행 파일
├── run_baseline.py                   # 경로 A 통계 baseline 실행
├── run_inference.py                  # 최종 추론 실행 (streaming + inference 연결)
├── requirements.txt
└── README.md
